Roblox Integratie

Voorbeeldcode voor Roblox

local HttpService = game:GetService("HttpService")

local API_KEY = "JOUW_API_SLEUTEL"
local API_URL = "JOUW_API_URL"

local function sendEmergencyReport(player, emergencyType, location, description)
	local url = API_URL .. "/api/meldingen"
    local url = API_URL .. "/api/meldingen"
    
    local headers = {
        ["Content-Type"] = "application/json",
        ["Authorization"] = "Bearer " .. API_KEY
    }
    
    local data = {
        playerName = player.Name,
        userId = player.UserId,
        emergencyType = emergencyType,
        location = {
            x = location.X,
            y = location.Y,
            z = location.Z
        },
        description = description,
        timestamp = os.time()
    }
    
    local success, response = pcall(function()
        return HttpService:PostAsync(url, HttpService:JSONEncode(data), Enum.HttpContentType.ApplicationJson, false, headers)
    end)
    
    return success, response
end

return {
    sendEmergencyReport = sendEmergencyReport
}

Plaats deze code in een ModuleScript en gebruik het om meldingen te versturen vanuit je Roblox game.
